🌐 Lesson 1: Intro to ESP8266 & ESP32

The ESP8266 and ESP32 are WiFi-capable microcontrollers from Espressif. They're much more powerful than the Arduino Uno and include built-in WiFi (and Bluetooth on ESP32). You program them using the Arduino IDE — same syntax, but with extra WiFi libraries.

Setting Up ESP8266/ESP32 in Arduino IDE

1
Open Preferences

In Arduino IDE: File → Preferences

2
Add Board Manager URL

In "Additional Boards Manager URLs", add:
ESP8266: http://arduino.esp8266.com/stable/package_esp8266com_index.json
ESP32: https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package_esp32_index.json

3
Install Board

Tools → Board → Boards Manager → Search "esp8266" or "esp32" → Install

4
Select Your Board

Tools → Board → ESP8266 Boards → NodeMCU 1.0 (or for ESP32: ESP32 Dev Module)

5
Select Port

Tools → Port → Select the COM port that appeared when you plugged in the board. If no port appears, install the CP2102 or CH340 driver.

💡
Which to use? For this module we'll use the ESP8266 NodeMCU for lessons 1–4 (easy to get, cheaper). Lesson 4 (BLE) uses ESP32 specifically. All ESP8266 code also works on ESP32 with only minor changes.

📶 Lesson 2: Connecting to WiFi

Connecting your ESP8266 to your home WiFi network is the first step to building IoT projects. Once connected, your device can fetch data from the internet, post sensor readings to servers, and receive commands remotely.

Arduino C++ (ESP8266)
#include <ESP8266WiFi.h>  // For ESP32 use: #include <WiFi.h>

const char* ssid     = "YOUR_WIFI_NAME";
const char* password = "YOUR_WIFI_PASSWORD";

void setup() {
  Serial.begin(115200);
  Serial.println("Connecting to WiFi...");

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("");
  Serial.println("✅ Connected!");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());
  Serial.print("Signal strength (RSSI): ");
  Serial.print(WiFi.RSSI());
  Serial.println(" dBm");
}

void loop() {
  // Check if still connected
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("WiFi lost! Reconnecting...");
    WiFi.reconnect();
  }
}
✅
Security tip: Never hard-code WiFi credentials in code you share publicly. For real projects, store credentials in a separate config file or use WiFiManager library to set them up via a captive portal.

🖥️ Lesson 3: Building a Web Server

Turn your ESP8266 into a tiny web server accessible from any browser on your local network. Control LEDs, read sensor values — all from a webpage!

Arduino C++ (ESP8266)
#include <ESP8266WiFi.h>
#include <ESP8266WebServer.h>

const char* ssid     = "YOUR_WIFI";
const char* password = "YOUR_PASS";

ESP8266WebServer server(80); // Port 80 = HTTP
#define LED_PIN 2  // Built-in LED on NodeMCU (active LOW)

bool ledState = false;

// HTML page sent to browser
String buildPage() {
  String html = "<!DOCTYPE html><html><head>";
  html += "<title>ESP8266 Control</title>";
  html += "<meta name='viewport' content='width=device-width,initial-scale=1'>";
  html += "<style>body{font-family:sans-serif;text-align:center;padding:30px}";
  html += "button{padding:15px 30px;font-size:18px;border-radius:8px;cursor:pointer;border:none}";
  html += ".on{background:#16a34a;color:white}.off{background:#dc2626;color:white}</style></head>";
  html += "<body><h1>⚡ AlCoders IoT Control</h1>";
  html += "<p>LED is currently: <strong>" + String(ledState ? "ON" : "OFF") + "</strong></p>";
  html += "<a href='/on'><button class='on'>Turn ON</button></a> ";
  html += "<a href='/off'><button class='off'>Turn OFF</button></a>";
  html += "<p>Uptime: " + String(millis()/1000) + "s</p></body></html>";
  return html;
}

void setup() {
  Serial.begin(115200);
  pinMode(LED_PIN, OUTPUT);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); }
  Serial.println("\nConnected! Visit: http://" + WiFi.localIP().toString());

  server.on("/", []() { server.send(200, "text/html", buildPage()); });
  server.on("/on", []() {
    ledState = true;
    digitalWrite(LED_PIN, LOW); // NodeMCU LED is active LOW
    server.sendHeader("Location", "/");
    server.send(302);
  });
  server.on("/off", []() {
    ledState = false;
    digitalWrite(LED_PIN, HIGH);
    server.sendHeader("Location", "/");
    server.send(302);
  });
  server.begin();
}

void loop() {
  server.handleClient(); // Check for incoming requests
}

Open your browser and type the IP address shown in Serial Monitor (e.g. 192.168.1.42). You'll see a web page with On/Off buttons that control the LED!

☁️ Lesson 4: Sending Data to the Cloud (ThingSpeak)

ThingSpeak is a free IoT cloud platform by MathWorks. You can post sensor data to it and see beautiful live graphs from anywhere in the world.

1
Create Account

Go to thingspeak.com → Sign up for free

2
Create Channel

New Channel → Add field names (e.g. "Temperature", "Humidity") → Save

3
Get API Key

In your channel: API Keys tab → Copy the "Write API Key"

Arduino C++ (ESP8266)
#include <ESP8266WiFi.h>
#include <ESP8266HTTPClient.h>
#include <DHT.h>  // Install "DHT sensor library" by Adafruit

const char* ssid     = "YOUR_WIFI";
const char* password = "YOUR_PASS";
String apiKey = "YOUR_THINGSPEAK_WRITE_KEY";
String server  = "http://api.thingspeak.com/update";

#define DHT_PIN  4  // D2 on NodeMCU
#define DHT_TYPE DHT11
DHT dht(DHT_PIN, DHT_TYPE);

void setup() {
  Serial.begin(115200);
  dht.begin();
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
  Serial.println("Connected!");
}

void loop() {
  float temp = dht.readTemperature();
  float hum  = dht.readHumidity();

  if (isnan(temp) || isnan(hum)) {
    Serial.println("DHT read error");
    return;
  }

  String url = server + "?api_key=" + apiKey
             + "&field1=" + String(temp)
             + "&field2=" + String(hum);

  WiFiClient client;
  HTTPClient http;
  http.begin(client, url);
  int code = http.GET();
  Serial.print("Temp:"); Serial.print(temp);
  Serial.print(" Hum:"); Serial.print(hum);
  Serial.print(" → HTTP:"); Serial.println(code);
  http.end();

  delay(20000); // ThingSpeak free tier: max 1 update per 15 seconds
}

📨 Lesson 5: MQTT Basics

MQTT (Message Queuing Telemetry Transport) is the standard protocol for IoT messaging. It uses a publish/subscribe model: devices publish messages to a "topic" and other devices subscribe to receive them. It's lightweight and perfect for Arduino/ESP devices.

MQTT Concepts

TermMeaningExample
BrokerServer that routes messagesbroker.hivemq.com (free public)
ClientAny device connecting to brokerYour ESP8266
TopicChannel/address for messageshome/livingroom/temperature
PublishSend a message to a topicESP8266 sends "25.3" to temperature topic
SubscribeReceive messages from a topicPhone app gets temperature updates
QoSDelivery guarantee level (0, 1, 2)QoS 0 = fire-and-forget
Arduino C++ (ESP8266)
// Install "PubSubClient" library by Nick O'Leary
#include <ESP8266WiFi.h>
#include <PubSubClient.h>

const char* ssid      = "YOUR_WIFI";
const char* password  = "YOUR_PASS";
const char* mqttServer = "broker.hivemq.com";

WiFiClient wifiClient;
PubSubClient mqtt(wifiClient);

// Called when a subscribed message arrives
void onMessage(char* topic, byte* payload, unsigned int length) {
  String msg = "";
  for (int i = 0; i < length; i++) msg += (char)payload[i];
  Serial.print("[MQTT] Topic: "); Serial.print(topic);
  Serial.print(" | Message: "); Serial.println(msg);

  if (msg == "on")  digitalWrite(LED_BUILTIN, LOW);
  if (msg == "off") digitalWrite(LED_BUILTIN, HIGH);
}

void connectMQTT() {
  while (!mqtt.connected()) {
    Serial.print("Connecting MQTT...");
    if (mqtt.connect("AlCodersESP8266")) {
      Serial.println("connected!");
      mqtt.subscribe("alcoders/led/control");
    } else {
      Serial.print("failed, rc="); Serial.println(mqtt.state());
      delay(2000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(LED_BUILTIN, OUTPUT);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
  mqtt.setServer(mqttServer, 1883);
  mqtt.setCallback(onMessage);
  connectMQTT();
}

void loop() {
  if (!mqtt.connected()) connectMQTT();
  mqtt.loop(); // Process incoming messages

  // Publish sensor reading every 10 seconds
  static unsigned long last = 0;
  if (millis() - last > 10000) {
    last = millis();
    String reading = String(analogRead(A0));
    mqtt.publish("alcoders/sensor/value", reading.c_str());
  }
}
