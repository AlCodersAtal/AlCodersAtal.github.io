Smart Weather Station

Build a weather station that reads temperature, humidity, atmospheric pressure and UV index, displays them on an OLED, logs to SD card, and uploads to the cloud via WiFi (ESP8266).

Hardware List

ComponentMeasuresProtocol
DHT22Temperature + Humidity1-Wire
BMP280Pressure + AltitudeI2C
GUVA-S12SDUV IndexAnalog
SSD1306 OLEDDisplayI2C
ESP8266 (NodeMCU)WiFiUART/AT
SD ModuleLocal loggingSPI
Arduino C++ — Weather Station Core
#include <DHT.h>
#include <Adafruit_BMP280.h>
#include <Adafruit_SSD1306.h>
#include <SD.h>

DHT dht(7, DHT22);
Adafruit_BMP280 bmp;
Adafruit_SSD1306 display(128, 64, &Wire, -1);

struct WeatherData {
  float temp, hum, pressure, altitude;
  int   uvRaw;
};

WeatherData readAll() {
  return {
    dht.readTemperature(),
    dht.readHumidity(),
    bmp.readPressure() / 100.0,   // Pa → hPa
    bmp.readAltitude(1013.25),
    analogRead(A0)                  // UV raw
  };
}

void displayData(WeatherData &w) {
  display.clearDisplay();
  display.setCursor(0,0);
  display.print("Temp: "); display.print(w.temp); display.println("C");
  display.print("Hum:  "); display.print(w.hum);  display.println("%");
  display.print("Pres: "); display.print(w.pressure); display.println("hPa");
  display.print("Alt:  "); display.print(w.altitude); display.println("m");
  display.display();
}

void logToSD(WeatherData &w) {
  File f = SD.open("weather.csv", FILE_WRITE);
  f.print(millis()); f.print(",");
  f.print(w.temp);  f.print(",");
  f.print(w.hum);   f.print(",");
  f.println(w.pressure);
  f.close();
}

void setup() {
  dht.begin(); bmp.begin();
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  SD.begin(10);
}

void loop() {
  WeatherData w = readAll();
  displayData(w);
  logToSD(w);
  delay(30000);   // Every 30 seconds
}

Obstacle-Avoiding Robot

Build a fully autonomous robot that drives forward, detects obstacles with an ultrasonic sensor, and steers around them — no remote control needed.

Hardware List

ComponentRole
Arduino UnoBrain
HC-SR04 ultrasonic sensorDistance measurement
SG90 ServoRotate sensor to look left/right
L298N motor driverDrive 2 DC motors
Robot chassis + DC motorsMovement
7.4V Li-ion batteryPower
Arduino C++ — Obstacle Avoiding Robot
#include <Servo.h>

// Ultrasonic
const int TRIG=8, ECHO=9;
// Motors
const int IN1=2,IN2=3,IN3=4,IN4=5,ENA=6,ENB=7;
const int SAFE_DIST = 20;   // cm — stop if closer

Servo sensorServo;

long getDistance() {
  digitalWrite(TRIG, LOW); delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  return pulseIn(ECHO, HIGH) / 58;  // µs → cm
}

void forward()  { digitalWrite(IN1,H); digitalWrite(IN2,L); digitalWrite(IN3,H); digitalWrite(IN4,L); }
void stopAll()  { digitalWrite(IN1,L); digitalWrite(IN2,L); digitalWrite(IN3,L); digitalWrite(IN4,L); }
void turnLeft() { digitalWrite(IN1,L); digitalWrite(IN2,H); digitalWrite(IN3,H); digitalWrite(IN4,L); }
void turnRight(){ digitalWrite(IN1,H); digitalWrite(IN2,L); digitalWrite(IN3,L); digitalWrite(IN4,H); }

void setup() {
  pinMode(TRIG,OUTPUT); pinMode(ECHO,INPUT);
  int pins[]={IN1,IN2,IN3,IN4,ENA,ENB};
  for(auto p:pins) pinMode(p,OUTPUT);
  analogWrite(ENA,180); analogWrite(ENB,180);
  sensorServo.attach(11);
  sensorServo.write(90); // Face forward
}

void loop() {
  long dist = getDistance();

  if (dist > SAFE_DIST) {
    forward();
  } else {
    stopAll();
    delay(300);

    // Look left and right
    sensorServo.write(0); delay(400);
    long distLeft = getDistance();
    sensorServo.write(180); delay(400);
    long distRight = getDistance();
    sensorServo.write(90);

    if (distLeft > distRight) {
      turnLeft(); delay(500);
    } else {
      turnRight(); delay(500);
    }
  }
}

Home Automation System

Control lights, fans, and appliances from anywhere using a web interface hosted directly on an ESP32. No internet required — connects to your home WiFi and creates a local dashboard accessible from any phone or browser.

Features

FeatureComponent
4-relay appliance control4-channel relay module
Temperature displayDHT22
Motion detection alertPIR sensor
Web dashboardESP32 web server
Schedule automationmillis() timer logic
ESP32 Arduino — Home Automation Web Server
#include <WiFi.h>
#include <WebServer.h>
#include <DHT.h>

const char* ssid     = "YourWiFi";
const char* password = "YourPassword";

WebServer server(80);
DHT dht(4, DHT22);

const int relays[] = {16,17,18,19};
bool relayState[4] = {false,false,false,false};

void handleRoot() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  String html = "<!DOCTYPE html><html><head><meta name='viewport' content='width=device-width,initial-scale=1'><title>AlCoders Home</title></head><body style='font-family:sans-serif;padding:20px'>";
  html += "<h1>🏠 Home Control</h1>";
  html += "<p>Temp: <b>" + String(t) + "°C</b> | Humidity: <b>" + String(h) + "%</b></p>";
  for(int i=0; i<4; i++) {
    html += "<p>Relay " + String(i+1) + ": <a href='/toggle?id=" + String(i) + "'><button>";
    html += relayState[i] ? "ON ✅" : "OFF ❌";
    html += "</button></a></p>";
  }
  html += "</body></html>";
  server.send(200, "text/html", html);
}

void handleToggle() {
  int id = server.arg("id").toInt();
  if(id >= 0 && id < 4) {
    relayState[id] = !relayState[id];
    digitalWrite(relays[id], relayState[id] ? LOW : HIGH);  // Relay = active LOW
  }
  server.sendHeader("Location", "/");
  server.send(303);
}

void setup() {
  for(int i=0; i<4; i++) { pinMode(relays[i], OUTPUT); digitalWrite(relays[i], HIGH); }
  dht.begin();
  WiFi.begin(ssid, password);
  while(WiFi.status() != WL_CONNECTED) delay(500);
  server.on("/", handleRoot);
  server.on("/toggle", handleToggle);
  server.begin();
}

void loop() { server.handleClient(); }

RFID Door Lock

Build a keycard-based door lock using an MFRC522 RFID reader. Tap your card → door opens. Wrong card → alarm sounds. Authorized UIDs stored in EEPROM so they persist after power-off.

Hardware List

ComponentRole
MFRC522 RFID readerRead card/tag UID
SG90 ServoLock/unlock mechanism
Green + Red LEDAccess granted/denied
BuzzerAlert tones
Push buttonAdd new authorized card
Arduino C++ — RFID Door Lock
#include <SPI.h>
#include <MFRC522.h>
#include <Servo.h>
#include <EEPROM.h>

#define SS_PIN   10
#define RST_PIN  9
#define LED_GREEN 6
#define LED_RED   7
#define BUZZER    5
#define ENROLL_BTN 4

MFRC522 rfid(SS_PIN, RST_PIN);
Servo lockServo;

// Store up to 5 authorized UIDs in EEPROM (4 bytes each)
const int MAX_CARDS = 5;
byte storedUIDs[MAX_CARDS][4];
int cardCount = 0;

bool isAuthorized(byte* uid) {
  for(int i=0; i<cardCount; i++) {
    if(memcmp(storedUIDs[i], uid, 4) == 0) return true;
  }
  return false;
}

void grantAccess() {
  lockServo.write(90);   // Unlock
  digitalWrite(LED_GREEN, HIGH);
  tone(BUZZER, 1000, 200);
  delay(3000);
  lockServo.write(0);    // Lock
  digitalWrite(LED_GREEN, LOW);
}

void denyAccess() {
  digitalWrite(LED_RED, HIGH);
  tone(BUZZER, 300, 1000);
  delay(1000);
  digitalWrite(LED_RED, LOW);
}

void loop() {
  if(!rfid.PICC_IsNewCardPresent() || !rfid.PICC_ReadCardSerial()) return;

  byte* uid = rfid.uid.uidByte;

  if(digitalRead(ENROLL_BTN) == LOW && cardCount < MAX_CARDS) {
    // Enroll mode — save new card
    memcpy(storedUIDs[cardCount], uid, 4);
    EEPROM.put(cardCount * 4, storedUIDs[cardCount]);
    cardCount++;
    tone(BUZZER, 2000, 500);
  } else {
    isAuthorized(uid) ? grantAccess() : denyAccess();
  }
  rfid.PICC_HaltA();
}

Project Framework — Use This Every Time

StepWhat to Do
1. DefineWhat does your project do? What inputs and outputs does it need?
2. Component ListList every sensor, actuator, display, and communication module
3. Circuit DesignDraw a schematic or Tinkercad diagram before wiring
4. Code StructurePlan setup(), loop(), and helper functions before writing
5. Build & TestBuild and test one component at a time — don't wire everything at once
6. IntegrateCombine components, test interactions
7. RefineAdd error handling, debouncing, power saving, and edge cases
