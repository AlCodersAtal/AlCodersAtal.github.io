Iterators & Generators

Generators produce values on-demand instead of storing them all in memory — perfect for large datasets, infinite sequences, or data pipelines.

Iterators

Python
# Any object with __iter__ and __next__ is an iterator nums = [1, 2, 3] it = iter(nums) # get iterator from list print(next(it)) # 1 print(next(it)) # 2 print(next(it)) # 3 # next(it) ← StopIteration! # Build a custom iterator class class CountUp: def __init__(self, stop): self.current = 0 self.stop = stop def __iter__(self): return self def __next__(self): if self.current >= self.stop: raise StopIteration self.current += 1 return self.current for n in CountUp(3): print(n) # 1 2 3

Generator Functions (yield)

Python
# yield turns a function into a generator — much simpler! def count_up(stop): for i in range(1, stop + 1): yield i # pauses here, resumes on next call for n in count_up(5): print(n) # Infinite generator — generates primes forever def infinite_primes(): n = 2 while True: if all(n % i != 0 for i in range(2, n)): yield n n += 1 # Get first 10 primes (never loads all into memory) import itertools print(list(itertools.islice(infinite_primes(), 10))) # Generator expression — like list comprehension but lazy big_sum = sum(x**2 for x in range(1_000_000)) # no list stored!

Decorators

Decorators wrap a function to add extra behaviour — logging, timing, caching, authentication — without modifying the original function.

How Decorators Work

Python
# A decorator is a function that wraps another function def my_decorator(func): def wrapper(*args, **kwargs): print("Before the function") result = func(*args, **kwargs) print("After the function") return result return wrapper # Apply with @ syntax @my_decorator def say_hello(name): print(f"Hello, {name}!") say_hello("Alice") # Before the function # Hello, Alice! # After the function

Practical Decorators

Python
import time import functools # Timer decorator def timer(func): @functools.wraps(func) # preserve docstring & name def wrapper(*args, **kwargs): start = time.time() result = func(*args, **kwargs) print(f"{func.__name__} took {time.time()-start:.3f}s") return result return wrapper # Cache/memoize decorator def cache(func): memo = {} @functools.wraps(func) def wrapper(*args): if args not in memo: memo[args] = func(*args) return memo[args] return wrapper @timer @cache def fib(n): return n if n < 2 else fib(n-1) + fib(n-2) # Python standard library has lru_cache built in: from functools import lru_cache @lru_cache(maxsize=128) def fib(n): return n if n < 2 else fib(n-1) + fib(n-2)

Concurrency — Threads & Async

Concurrency lets your program do multiple things at once. Python offers threading for I/O-bound tasks and asyncio for efficient async programming.

Threading

Python
import threading import time def download(url): print(f"Downloading {url}...") time.sleep(2) # simulate network wait print(f"Done: {url}") urls = ["site1.com", "site2.com", "site3.com"] # Without threads — takes 6 seconds (2s × 3) # for url in urls: # download(url) # With threads — takes ~2 seconds (all run in parallel) threads = [threading.Thread(target=download, args=(url,)) for url in urls] for t in threads: t.start() for t in threads: t.join() # wait for all threads

asyncio — Modern Async Python

Python
import asyncio async def fetch(url): print(f"Fetching {url}") await asyncio.sleep(2) # non-blocking wait print(f"Done: {url}") return f"data from {url}" async def main(): # Run all fetches concurrently results = await asyncio.gather( fetch("site1.com"), fetch("site2.com"), fetch("site3.com"), ) print(results) asyncio.run(main())

Type Hints

Type hints (Python 3.5+) annotate variables and function signatures with expected types. They don't enforce types at runtime but power IDE autocomplete, static analysis, and team collaboration.

Basic Annotations

Python
from typing import Optional, Union, List, Dict, Tuple # Variable annotations name: str = "Alice" age: int = 25 scores: list[float] = [] # Function signatures def greet(name: str, times: int = 1) -> str: return f"Hello, {name}! " * times # Optional — value can be that type or None def find_user(user_id: int) -> Optional[dict]: return None # or a dict # Union — one of several types (Python 3.10+: int | str) def process(value: Union[int, str]) -> str: return str(value) # Complex types def get_stats(data: list[float]) -> dict[str, float]: return {"mean": sum(data)/len(data), "max": max(data)}

Testing with pytest

Tests verify that your code works correctly — and keeps working as you change it. pytest is the industry-standard Python testing framework.

Writing Tests

math_utils.py
def add(a, b): return a + b def divide(a, b): if b == 0: raise ValueError("Cannot divide by zero") return a / b
test_math_utils.py
import pytest from math_utils import add, divide # Test functions must start with test_ def test_add_integers(): assert add(2, 3) == 5 def test_add_floats(): assert add(0.1, 0.2) == pytest.approx(0.3) def test_add_negatives(): assert add(-1, -1) == -2 def test_divide_normal(): assert divide(10, 2) == 5.0 def test_divide_by_zero(): # Test that the right exception is raised with pytest.raises(ValueError, match="Cannot divide by zero"): divide(5, 0) # Parameterised tests — test many inputs at once @pytest.mark.parametrize("a, b, expected", [ (1, 1, 2), (0, 5, 5), (-3, 3, 0), ]) def test_add_parametrized(a, b, expected): assert add(a, b) == expected
Terminal
pip install pytest pytest # run all tests pytest -v # verbose output pytest test_math_utils.py # specific file pytest -k "divide" # tests matching pattern

Functional Tools & Builtins

Python's built-in functional tools — map, filter, zip, enumerate, any, all — let you write expressive, concise code without explicit loops.

Power Builtins

Python
nums = [3, 1, 4, 1, 5, 9, 2, 6] # enumerate — index + value for i, val in enumerate(nums): print(f"Index {i}: {val}") # zip — combine iterables names = ["Alice", "Bob", "Charlie"] scores = [92, 78, 85] for name, score in zip(names, scores): print(f"{name}: {score}") # any / all — boolean on iterables print(any(x > 8 for x in nums)) # True — some > 8 print(all(x > 0 for x in nums)) # True — all > 0 # map & filter (generators) doubled = list(map(lambda x: x*2, nums)) positives = list(filter(lambda x: x>0, [-1, 2, -3, 4])) # reduce — fold a list into one value from functools import reduce product = reduce(lambda x, y: x*y, [1, 2, 3, 4]) # 24 # sorted with key words = ["banana", "apple", "cherry"] print(sorted(words, key=len)) # by length print(sorted(words, key=lambda w: w[-1])) # by last letter
