Project 1: CLI To-Do App

Build a command-line to-do list manager that saves tasks to a JSON file. Covers file I/O, JSON, OOP, and user input handling.

Features

  • Add, list, complete, and delete tasks
  • Persist tasks to todos.json between runs
  • Colour-coded output for pending vs done
  • Command-line arguments with argparse

Build Steps

  1. Define a Task dataclass with id, title, done, created_at
  2. Define a TodoApp class with load(), save(), add(), list(), complete(), delete()
  3. Implement JSON persistence using json.dump / json.load
  4. Build a CLI with argparse subcommands: add, list, done, delete
  5. Add colour output using ANSI escape codes or colorama
todo.py — Core structure
from dataclasses import dataclass, field, asdict from datetime import datetime import json, argparse from pathlib import Path @dataclass class Task: id: int title: str done: bool = False created_at: str = field(default_factory=lambda: datetime.now().isoformat()) class TodoApp: FILE = Path("todos.json") def __init__(self): self.tasks: list[Task] = self.load() def load(self) -> list[Task]: if not self.FILE.exists(): return [] data = json.loads(self.FILE.read_text()) return [Task(**t) for t in data] def save(self): self.FILE.write_text(json.dumps([asdict(t) for t in self.tasks], indent=2)) def add(self, title: str): new_id = max((t.id for t in self.tasks), default=0) + 1 self.tasks.append(Task(id=new_id, title=title)) self.save() print(f"✓ Added: [{new_id}] {title}") if __name__ == "__main__": parser = argparse.ArgumentParser(description="CLI To-Do App") subs = parser.add_subparsers(dest="command") add_p = subs.add_parser("add") add_p.add_argument("title") subs.add_parser("list") # ... more subcommands args = parser.parse_args() app = TodoApp() if args.command == "add": app.add(args.title)

Project 2: File Organiser

Build a script that scans a directory and automatically sorts files into subfolders by type — images, documents, videos, code, etc.

Features

  • Detect file types by extension
  • Move files into categorised subfolders
  • Generate a summary report of what was moved
  • Dry-run mode to preview without moving files
  • Undo option using a log file
organiser.py
from pathlib import Path import shutil, json from collections import defaultdict CATEGORIES = { "Images": [".jpg", ".jpeg", ".png", ".gif", ".svg"], "Documents": [".pdf", ".docx", ".txt", ".xlsx"], "Videos": [".mp4", ".mkv", ".mov"], "Audio": [".mp3", ".wav", ".flac"], "Code": [".py", ".js", ".html", ".css"], } def get_category(ext: str) -> str: for cat, exts in CATEGORIES.items(): if ext.lower() in exts: return cat return "Other" def organise(folder: str, dry_run: bool = False): src = Path(folder) log = [] summary = defaultdict(int) for file in src.iterdir(): if not file.is_file(): continue cat = get_category(file.suffix) dest_dir = src / cat dest = dest_dir / file.name print(f"{'[DRY]' if dry_run else 'MOVE'}: {file.name} → {cat}/") if not dry_run: dest_dir.mkdir(exist_ok=True) shutil.move(str(file), str(dest)) log.append({"from": str(file), "to": str(dest)}) summary[cat] += 1 if not dry_run: Path("organiser_log.json").write_text(json.dumps(log, indent=2)) print("\nSummary:", dict(summary))

Project 3: Web Scraper

Build a web scraper that fetches data from a website, parses HTML, and exports the results to CSV or JSON.

Setup

Terminal
pip install requests beautifulsoup4

The Scraper

scraper.py
import requests from bs4 import BeautifulSoup import csv, json, time def scrape_quotes(pages: int = 3) -> list[dict]: """Scrape quotes from quotes.toscrape.com (a practice site).""" base = "http://quotes.toscrape.com/page/{}/" results = [] for page in range(1, pages + 1): print(f"Scraping page {page}...") resp = requests.get(base.format(page)) soup = BeautifulSoup(resp.text, "html.parser") for quote in soup.select(".quote"): results.append({ "text": quote.select_one(".text").text.strip("\u201c\u201d"), "author": quote.select_one(".author").text, "tags": [t.text for t in quote.select(".tag")], }) time.sleep(0.5) # be polite to the server return results def save_csv(data, filename="quotes.csv"): with open(filename, "w", newline="") as f: w = csv.DictWriter(f, fieldnames=["text", "author", "tags"]) w.writeheader() w.writerows(data) print(f"Saved {len(data)} quotes to {filename}") if __name__ == "__main__": quotes = scrape_quotes(pages=3) save_csv(quotes)

Project 4: REST API Client

Build a polished command-line weather app using a live REST API. Covers HTTP requests, JSON parsing, error handling, and beautiful terminal output.

Setup

Terminal
pip install requests # Get a free API key from: https://openweathermap.org/api
weather.py
import requests, os, sys from datetime import datetime API_KEY = os.getenv("WEATHER_API_KEY") BASE = "https://api.openweathermap.org/data/2.5/weather" def get_weather(city: str) -> dict: params = {"q": city, "appid": API_KEY, "units": "metric"} resp = requests.get(BASE, params=params, timeout=10) resp.raise_for_status() return resp.json() def display(data: dict): city = data["name"] country = data["sys"]["country"] temp = data["main"]["temp"] feels = data["main"]["feels_like"] desc = data["weather"][0]["description"].capitalize() humidity = data["main"]["humidity"] wind = data["wind"]["speed"] print(f""" ┌─────────────────────────────┐ │ 🌍 {city}, {country:20}│ │ 🌡️ {temp:.1f}°C (feels {feels:.1f}°C) │ 💧 Humidity: {humidity}% │ 🌬️ Wind: {wind} m/s │ ☁️ {desc} └─────────────────────────────┘""") if __name__ == "__main__": city = " ".join(sys.argv[1:]) or input("Enter city: ") try: data = get_weather(city) display(data) except requests.HTTPError as e: print(f"Error: City not found or API issue: {e}") except requests.ConnectionError: print("No internet connection.")
